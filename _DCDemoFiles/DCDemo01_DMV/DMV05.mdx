		SELECT
			[CATALOG_NAME],
			[CUBE_NAME],
			[DIMENSION_UNIQUE_NAME],
			[HIERARCHY_UNIQUE_NAME],
			[LEVEL_NAME],
			[LEVEL_UNIQUE_NAME],
			[LEVEL_NUMBER],
			[LEVEL_CAPTION],
			[LEVEL_CARDINALITY],
			[LEVEL_NAME_SQL_COLUMN_NAME],
			[LEVEL_KEY_SQL_COLUMN_NAME]
		FROM $SYSTEM.MDSCHEMA_LEVELS
		WHERE 1=1
		AND [CUBE_NAME] = 'OperationA'
		AND [DIMENSION_UNIQUE_NAME] = '[Geography]'
		AND [LEVEL_NUMBER]>0
		AND [LEVEL_IS_VISIBLE]
