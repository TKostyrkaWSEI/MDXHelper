		SELECT
			[CATALOG_NAME],
			[CUBE_NAME],
			[DIMENSION_NAME],
			[DIMENSION_UNIQUE_NAME],
			[DIMENSION_MASTER_NAME],
			[DIMENSION_CARDINALITY],
			[DIMENSION_TYPE],
			[DEFAULT_HIERARCHY]
		FROM $SYSTEM.MDSCHEMA_DIMENSIONS
		WHERE 1=1
		AND [CUBE_NAME] = 'OperationA'
		AND [DIMENSION_IS_VISIBLE]
		ORDER BY
			[DIMENSION_TYPE]
